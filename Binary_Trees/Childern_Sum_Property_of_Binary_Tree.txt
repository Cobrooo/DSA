// Only Function to maintain Children Sum Property in Binary Tree



class Solution {
    public static void childSum(Node root) {
        if(root == null) {
            return;
        }
        int child = 0;
        if(root.left != null) {
            child += root.left.val;
        }
        if(root.right != null) {
            child += root.right.val;
        }
        if(child >= root.val) {
            root.val = child;
        }
        else {
            if(root.left != null) {
                root.left.val = child;
            }
            else if(root.right != null) {
                root.right.val = child;
            }
        }
        childSum(root.left);
        childSum(root.right);
        int tot = 0;
        if(root.left != null) {
            tot += root.left.val;
        }
        if(root.right != null) {
            tot += root.right.val;
        }
        if(root.left != null || root.right != null) {
            root.val = tot;
        }
    }
}
