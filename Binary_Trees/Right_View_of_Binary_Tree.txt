// Only Function for the Right View of Binary Tree


class Pair {
    Node node;
    int y;
    Pair(Node node, int y) {
        this.node = node;
        this.y = y;
    }
}


class Solution {
    public static List<Integer> rightView(Node root) {
        List<Integer> ans = new ArrayList<>();
        if(root == null) {
            return ans;
        }
        Deque<Pair> q = new ArrayDeque<>();
        TreeMap<Integer, Integer> tm = new TreeMap<>();
        q.add(new Pair(root, 0));
        while(!q.isEmpty()) {
            Pair curr = q.poll();
            Node node = curr.node;
            int y = curr.y;
            tm.put(x, node.val);
            if(node.left != null) {
                q.add(new Pair(node.left, y+1));
            }
            if(node.right != null) {
                q.add(new Pair(node.right, y+1));
            }
        }
        for(int val : tm.values()) {
            ans.add(val);
        }
        return ans;
    }
}
