// Only function for thr Morris Inorder Traversal (Space Optimization)


class Solution {
    public static List<Integer> getInorder(TreeNode root) {
        List<Integer> ans = new ArrayList<>();
        TreeNode curr = root;
        while(curr != null) {
            if(curr.left == null) {
                ans.add(curr.val);
                curr = curr.right;
            }
            else {
                TreeNode rightMost = curr;
                while(rightMost.right != null && rightMost.right != curr) {
                    rightMost = rightMost.right;
                }
                if(rightMost.right == null) {
                    rightMost.right = curr;
                    curr = curr.left;
                }
                else {
                    rightMost.right = null;
                    ans.add(curr.val);
                    curr = curr.right;
                }
            }
        }
        return ans;;
    }
}
