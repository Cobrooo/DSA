// Code to Detect a Cycle in given Undirected Graph using BFS






import java.util.*;
class Main {
    public static boolean cycleBFS(List<List<Integer>> graph, boolean visited[], int start) {
        Deque<Pair> q = new ArrayDeque<>();
        q.add(new Pair(start, -1));
        visited[start] = true;
        while(!q.isEmpty()) {
            Pair curr = q.poll();
            int node = curr.node;
            int parent = curr.parent;
            for(int i=0;i<graph.get(node).size();i++) {
                int val = graph.get(node).get(i);
                if(!visited[val]) {
                    q.add(new Pair(val, node));
                    visited[val] = true;
                }
                else if(parent != val) {
                    return true;
                }
            }
        }
        return false;
    }
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int e = sc.nextInt();
        List<List<Integer>> graph = new ArrayList<>();
        for(int i=0;i<=n;i++) {
            graph.add(new ArrayList<>());
        }
        for(int i=0;i<e;i++) {
            int u = sc.nextInt();
            int v = sc.nextInt();
            graph.get(u).add(v);
            graph.get(v).add(u);
        }
        boolean visited[] = new boolean[n+1];
        System.out.println(cycleBFS(graph, visited, 1));
        sc.close();
    }
}
class Pair {
    int node;
    int parent;
    Pair(int node, int parent) {
        this.node = node;
        this.parent = parent;
    }
}
