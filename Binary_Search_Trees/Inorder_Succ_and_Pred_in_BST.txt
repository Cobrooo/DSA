// Only function to find the Inorder successor and predcessor in Binary Search Tree




class Solution {
    List<Integer> succPredBST(TreeNode root, int key) {
        TreeNode successor = null;
        TreeNode predecessor = null;
        TreeNode curr = root;
        while(curr != null) {
            if(curr.data <= key) {
                curr = curr.right;
            }
            else {
                successor = curr;
                curr = curr.left;
            }
        }
        curr = root;
        while(curr != null) {
            if(curr.data < key) {
                predecessor = curr;
                curr = curr.right;
            }
            else {
                curr = curr.left;
            }
        }
        int pred = (predecessor != null) ? predecessor.data : -1;
        int succ = (successor != null) ? successor.data : -1;
        return Arrays.asList(pred, succ);
    }
}
